      Base URL: (click to copy)

      POST /api/generate-soul

      # Genera una singola AI Soul
curl -X POST /api/generate-soul \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Marco",
    "role": "Full Stack Developer",
    "traits": "curioso, resiliente",
    "values": "liberta, artigianato"
  }'

      POST /api/generate-team

      # Genera N souls in parallelo + analisi sinergie
curl -X POST /api/generate-team \
  -H "Content-Type: application/json" \
  -d '{
    "team_name": "Astra Core",
    "members": [
      {"name": "Mattia", "role": "CEO", "traits": "visionario, forgiatore"},
      {"name": "Marco", "role": "CTO", "traits": "pragmatico, designer"},
      {"name": "Roberto", "role": "Dev", "traits": "leale, veloce"}
    ]
  }'
